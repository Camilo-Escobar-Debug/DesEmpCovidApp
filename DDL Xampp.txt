Create table login(
id_login BIGINT UNSIGNED NOT NULL AUTO_INCREMENT,
usuario VARCHAR(10) not null,
clave VARCHAR(25),
PRIMARY KEY(id_login));

Create table enfermedades(
id_enfermedad BIGINT UNSIGNED NOT NULL AUTO_INCREMENT,
descripcion VARCHAR(20) not null,
PRIMARY KEY(id_enfermedad));

Create table decretos(
id_decreto BIGINT UNSIGNED NOT NULL AUTO_INCREMENT,
numero_decreto VARCHAR(10) not null,
descripcion VARCHAR(500) not null,
PRIMARY KEY(id_decreto));

Create table tipos_documentos(
id_tipo_documento BIGINT UNSIGNED NOT NULL AUTO_INCREMENT,
descripcion VARCHAR(15) not null,
PRIMARY KEY(id_tipo_documento));

Create table departamentos(
id_departamento BIGINT UNSIGNED NOT NULL AUTO_INCREMENT,
nombre VARCHAR(50) not null,
PRIMARY KEY(id_departamento));

Create table ciudades(
id_ciudad BIGINT UNSIGNED NOT NULL AUTO_INCREMENT,
id_departamento BIGINT UNSIGNED NOT NULL,
nombre VARCHAR(20),
PRIMARY KEY(id_ciudad));

Create table empresas_logins(
id_empresa_login BIGINT UNSIGNED NOT NULL AUTO_INCREMENT,
id_login BIGINT UNSIGNED NOT NULL,
id_empresa BIGINT UNSIGNED NOT NULL,
fecha_creacion DATETIME not null,
fecha_cancelado DATETIME not null,
estado INT not null,
PRIMARY KEY(id_empresa_login));

Create table empresas(
id_empresa BIGINT UNSIGNED NOT NULL AUTO_INCREMENT,
nombres VARCHAR(50) not null,
nit numeric(15) not null,
numero_verificacion INT not null,
representante VARCHAR(50), 
direccion VARCHAR(50),
email VARCHAR(80) not null,
telefono numeric(10) not null,
cantidad_empleados INT,
PRIMARY KEY(id_empresa));

Create table personas(
id_persona BIGINT UNSIGNED NOT NULL AUTO_INCREMENT,
id_tipo_documento BIGINT UNSIGNED NOT NULL,
id_ciudad BIGINT UNSIGNED NOT NULL,
documento VARCHAR(15) not null,
primer_apellido VARCHAR(20) not null,
segundo_apellido VARCHAR(20) not null,
primer_nombre VARCHAR(20) not null,
segundo_nombre VARCHAR(20) not null,
fecha_nacimiento DATE not null,
estatura INT,
grupo_sanguineo varchar(3),
genero SMALLINT not null,
peso INT,
telefono numeric(12) not null,
email VARCHAR(100) not null,
PRIMARY KEY(id_persona));

Create table registros(
id_registro BIGINT UNSIGNED NOT NULL AUTO_INCREMENT,
id_persona BIGINT UNSIGNED NOT NULL,
id_empresa BIGINT UNSIGNED NOT NULL,
id_enfermedad BIGINT UNSIGNED NOT NULL,
fecha_ingreso DATETIME not null,
fecha_salida DATETIME not null,
temperatura int not null,
sintomas BOOLEAN,
ingreso BOOLEAN,
PRIMARY KEY(id_registro));

Create table historicos_decretos(
id_historico BIGINT UNSIGNED NOT NULL AUTO_INCREMENT,
id_ciudad BIGINT UNSIGNED NOT NULL,
id_decreto BIGINT UNSIGNED NOT NULL,
fecha_decreto DATE not null,
descripcion VARCHAR(1000) not null,
url VARCHAR(500) not null,
PRIMARY KEY(id_historico));
